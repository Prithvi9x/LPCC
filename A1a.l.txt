%{
#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX 1000

char wordsA[MAX][50];
int countA = 0, match = 0;

char *stopwords[] = {"is","am","a","the","if","in","on","at","of","to",NULL};

int is_stopword(char *word) {
    for(int i=0; stopwords[i]!=NULL; i++) {
        if(strcmp(word, stopwords[i]) == 0)
            return 1;
    }
    return 0;
}

void to_lower(char *word){
    for(int i=0; word[i]; i++)
        word[i] = tolower(word[i]);
}
%}
