%{
#include "y.tab.h"
#include <stdlib.h>
#include <string.h>
%}

%%

[0-9]+(\.[0-9]+)? {
    yylval.dval = atof(yytext);
    return NUMBER;
}

"sin"   return SIN;
"cos"   return COS;
"sqrt"  return SQRT;
"log"   return LOG;

[a-zA-Z][a-zA-Z0-9]* {
    yylval.sval = strdup(yytext);
    return NAME;
}

[ \t]   ;
\n      return '\n';

.       return yytext[0];

%%

int yywrap() {
    return 1;
}
